<template>
  <div id="app">
    <form v-on:submit.prevent="reverseMessage(message)">
      <input type="text" v-model="message" />
      <button type="submit">Add</button>
    </form>
    <hr />
    <ul>
      <li v-for="(todo, i) in array" :key="i">
        {{ i + 1 }}-{{ todo }}
        <button v-on:click="SliceMessage(i)">create</button>
      </li>
    </ul>
  </div>
</template>

<script>
export default {
  name: "TodoItem",
  // ...
  data() {
    return {
      message: "Example ",
      array: [" TodoExample", "TodoItems"],
    };
  },
  methods: {
    reverseMessage(message) {
      if (this.message != "") {
        this.array.push(message);
        this.message = "";
      }
    },
    SpliceMessage(i) {
      this.array.splice(i, 1);
    },
  },
};
</script>
